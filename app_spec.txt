<project_specification>
  <project_name>OmniWriter</project_name>

  <overview>
    OmniWriter è una piattaforma di scrittura professionale basata su intelligenza artificiale che permette di generare romanzi, saggi e articoli giornalistici attraverso tre macroaree specializzate ("Romanziere", "Saggista", "Redattore"). L'applicazione sfrutta l'orchestrazione di più modelli AI e l'addestramento su stili umani personalizzati (Human Model) per produrre testi complessi e coerenti, basati su fonti caricate dall'utente o ricerche web in tempo reale. È pensata per scrittori, esperti/divulgatori, giornalisti e professionisti della comunicazione.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React</framework>
      <styling>Tailwind CSS + shadcn/ui</styling>
      <ai_integration>Vercel AI SDK</ai_integration>
      <language>TypeScript</language>
      <i18n>Italiano + Inglese (i18next)</i18n>
    </frontend>
    <backend>
      <runtime>Node.js</runtime>
      <framework>Express.js</framework>
      <database>SQLite (via better-sqlite3 or Drizzle ORM)</database>
      <auth>Passport.js (email/password + Google OAuth)</auth>
      <file_upload>Multer (PDF, DOCX, RTF, TXT)</file_upload>
    </backend>
    <communication>
      <api>REST API</api>
      <realtime>Server-Sent Events (SSE) for AI generation streaming</realtime>
    </communication>
  </technology_stack>

  <database_schema>
    <tables>
      <users>
        - id (PRIMARY KEY), email (UNIQUE), password_hash, name, bio, avatar_url
        - role (free/premium/lifetime/admin), subscription_status, subscription_expires_at
        - preferred_language (it/en), theme_preference (light/dark)
        - google_id (UNIQUE, nullable), created_at, updated_at, last_login_at
      </users>

      <sessions>
        - id (PRIMARY KEY), user_id (FK users), token, expires_at, created_at
      </sessions>

      <projects>
        - id (PRIMARY KEY), user_id (FK users), saga_id (FK sagas, nullable)
        - title, description, area (romanziere/saggista/redattore)
        - genre, tone, target_audience, pov, word_count_target
        - status (draft/in_progress/completed/archived)
        - human_model_id (FK human_models, nullable)
        - settings_json (area-specific configuration)
        - word_count, created_at, updated_at
      </projects>

      <sagas>
        - id (PRIMARY KEY), user_id (FK users), title, description
        - area (romanziere/saggista/redattore), created_at, updated_at
      </sagas>

      <chapters>
        - id (PRIMARY KEY), project_id (FK projects), title, content
        - order_index, status (draft/generated/revised/final)
        - word_count, created_at, updated_at
      </chapters>

      <chapter_versions>
        - id (PRIMARY KEY), chapter_id (FK chapters), content, version_number
        - created_at, change_description
      </chapter_versions>

      <characters>
        - id (PRIMARY KEY), project_id (FK projects), saga_id (FK sagas, nullable)
        - name, description, traits, backstory, role_in_story
        - relationships_json, extracted_from_upload (boolean), created_at, updated_at
      </characters>

      <locations>
        - id (PRIMARY KEY), project_id (FK projects), saga_id (FK sagas, nullable)
        - name, description, significance
        - extracted_from_upload (boolean), created_at, updated_at
      </locations>

      <plot_events>
        - id (PRIMARY KEY), project_id (FK projects), saga_id (FK sagas, nullable)
        - title, description, chapter_id (FK chapters, nullable)
        - order_index, event_type, extracted_from_upload (boolean)
        - created_at, updated_at
      </plot_events>

      <human_models>
        - id (PRIMARY KEY), user_id (FK users), name, description
        - model_type (romanziere_advanced/saggista_basic/redattore_basic)
        - analysis_result_json (tone, sentence_structure, vocabulary, patterns)
        - total_word_count, training_status (pending/analyzing/ready/failed)
        - style_strength (0-100), created_at, updated_at
      </human_models>

      <human_model_sources>
        - id (PRIMARY KEY), human_model_id (FK human_models)
        - file_name, file_path, file_type, word_count
        - uploaded_at
      </human_model_sources>

      <sources>
        - id (PRIMARY KEY), project_id (FK projects, nullable), saga_id (FK sagas, nullable)
        - user_id (FK users), file_name, file_path, file_type, file_size
        - content_text (extracted), source_type (upload/web_search)
        - url (nullable, for web sources), tags_json
        - relevance_score, created_at
      </sources>

      <generation_logs>
        - id (PRIMARY KEY), project_id (FK projects), chapter_id (FK chapters, nullable)
        - model_used, phase (structure/writing/revision)
        - tokens_input, tokens_output, duration_ms
        - status (started/completed/failed/cancelled)
        - error_message (nullable), created_at
      </generation_logs>

      <project_tags>
        - id (PRIMARY KEY), project_id (FK projects), tag_name
      </project_tags>

      <export_history>
        - id (PRIMARY KEY), project_id (FK projects), format (docx/epub/rtf/pdf/txt)
        - file_path, epub_cover_url (nullable), epub_metadata_json (nullable)
        - created_at
      </export_history>

      <user_preferences>
        - id (PRIMARY KEY), user_id (FK users, UNIQUE)
        - default_ai_model, default_quality_setting (speed/balanced/quality)
        - dashboard_layout_json, keyboard_shortcuts_json
        - updated_at
      </user_preferences>
    </tables>
  </database_schema>
